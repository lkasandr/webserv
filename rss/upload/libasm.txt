; Безусловный прыжок - прыгаем без условий:

jmp	label	; следующие переходы зависят от флагов оставленных в cmp

;Условный прыжок - прыгаем только если соблюдается условие
je	label	; Jump if equal (прыгаем если равно)
jne label ; Jump if not equeal (прыгаем если неравно)
jl label ; Jump if less (прыгаем если меньше < VALUE)
jle label ; Jump if less or equal (прыгаем если меньше/равно <= VALUE)
jz label ; Jump if zero (прыгаем если ноль == 0)
jnz label ; Jump if not zero (прыгаем если не ноль != 0)
jg label ; Jump if greater (прыгаем если больше >)
jge label ; Jump if greater or equal (прыгаем если больше/равно >= VALUE)


   

mov x, y      	x ← y
and x, y		x ← x and y
or x, y		x ← x or y
xor x, y		x ← x xor y
add x, y		x ← x + y
sub x, y		x ← x – y
inc x			x ← x + 1
dec x			x ← x – 1
syscall		Invoke an operating system routine
db			A pseudo-instruction that declares bytes that will be in memory when the program runs
